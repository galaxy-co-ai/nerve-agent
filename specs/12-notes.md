  contentPlain: string         // For search
  folder: NoteFolder
  tags: NoteTag[]
  links: NoteLink[]            // Outgoing links
  backlinks: NoteLink[]        // Incoming links

  linkedItems: LinkedItem[]    // Tasks, decisions, etc.

  updatedAt: Date
  accessedAt: Date
  wordCount: number
  isArchived: boolean

  fromNote: Note
  toNote: Note
  context: string              // Surrounding text
}

  id: string
  name: string
  parent?: NoteFolder
  isAutoGenerated: boolean
  sortOrder: number
